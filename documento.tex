\section{Introdução}

\begin{frame}
	\frametitle{CSS}
	
	\begin{itemize}
		\item Um acrônimo para \underline{C}ascading \underline{S}tyle \underline{S}heets;
		\item Proposto por Håkon Wium Lie e alançado oficialmente em 1996;
		\item É usada para definir a aparência visual de documentos HTML;
		\item Controla layout, cores, fontes e espaçamentos, separando o conteúdo da estrutura HTML da sua apresentação; e
		\item Permite criar sites responsivos, organizados e consistentes.
	\end{itemize}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Regra de estilo}
	
	\begin{block}{Estrutura}
		\begin{lstlisting}
seletor {
	propriedade: valor;
}
		\end{lstlisting}
	\end{block}\vfill
	
	\begin{itemize}
		\item O \structure{seletor} define o alvo da regra de estilo;
		\item A \structure{propriedade} define o que será utilizado; e
		\item O \structure{valor} define como será utilizado.
	\end{itemize} \vfill
	
	\begin{exampleblock}{Exemplo}
	\begin{lstlisting}
p {
   color: blue;
}
	\end{lstlisting}
\end{exampleblock}
\end{frame}

\section{Folha de estilo}

\begin{frame}
\frametitle{Folha de estilo}

\begin{itemize}
	\item Conjunto de regras de estilo aplicado a um site;
	\item Existem três formas de declarar:
	\begin{enumerate}
		\item Inline;
		\item Incorporado; e
		\item Externo.
	\end{enumerate} 
\end{itemize}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Inline}
	
	\begin{itemize}
		\item As regras de estilo são declaradas diretamente nas tags; e
		\item Utiliza o atributo \structure{style} para definir o estilo desejado.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
<body style="background-color=yellow">

</body>
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Incorporado}
	
	\begin{itemize}
		\item A tag \structure{<style>} é usada para declarar a folha de estilo; e
		\item Deve estar ao final da tag \structure{<head>}.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
<head>
   ...
   <style>
      body { background-color: yellow; }
      p { 
      	  color: blue; 
          font-size: 15pt; 
         }
   </style>
</head>
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Externo}
	
	\begin{itemize}
		\item Dever ser criado um arquivo com extensão \structure{.css};
		\item Dentro do arquivo criado são definidas as regras de estilo; e
		\item O arquivo de estilo é importando no documento HTML usando a tag \structure{<link>}, que deve estar dentro da tag \structure{<head>}.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
<head>
   <link rel="stylesheet" type="text/css" href="estilo.css" 
    media="all">
</head>
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\section{Seletores}

\begin{frame}
	\frametitle{Tipos de seletores}
	
	\begin{itemize}
		\item Classe;
		\item ID;
		\item Universal;
		\item Atributo; e
		\item Pseudoclasse.
	\end{itemize}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Seletor de classe}
	
	\begin{itemize}
		\item Seleciona elementos com um atributo class específico, definido com um ponto \structure{.} ; e
		\item Pode ser utilizado em um ou mais elementos.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
.caixa{
   border: 1px solid green;
   background-color: yellow;
   color: green;
}

<p class="caixa"></p>			
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Seletor ID}
	
	\begin{itemize}
		\item Seleciona um único elemento com um atributo id único, definido com hashtag \structure{\#} ; e
		\item \alert{Não pode ser usado em outro elemento}.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
#cabecalho{
   border: 1px dashed blue;
   background-color: gray;
   color: blue;
}

<h1 id="cabecalho"></h1>			
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Seletor universal}
	
	\begin{itemize}
		\item Utiliza o asterisco \structure{*} para aplicar estilos a todos os elementos da página.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
*{
   box-sizing: border-box;
}			
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Seletor de atributo}
	
	\begin{itemize}
		\item Seleciona os elementos com base em seus atributos ou valores.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
input[type="text"]{
   background-color: blue;
   color: white;
}			
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Seletor de pseudoclasse}
	
	\begin{itemize}
		\item Selecionam elementos com base no seu estado, utilizando dois pontos \structure{:} .
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
a:hover{
   background-color: blue;
   color: white;
}			
		\end{lstlisting}
	\end{exampleblock}
\end{frame}
